001    package realtimeweb.magicthegatheringservice.structured;
002    
003    import realtimeweb.magicthegatheringservice.main.AbstractMagicTheGatheringService;
004    import realtimeweb.magicthegatheringservice.util.Util;
005    
006    import java.util.ArrayList;
007    import java.util.HashMap;
008    import java.util.LinkedHashMap;
009    import com.google.gson.Gson;
010    import realtimeweb.magicthegatheringservice.json.JsonMagicTheGatheringService;
011    import realtimeweb.magicthegatheringservice.json.JsonGetCardListener;
012    import realtimeweb.magicthegatheringservice.json.JsonSearchCardsListener;
013    
014    /**
015     * Used to get data as built-in Java objects (HashMap, ArrayList, etc.).
016     */
017    public class StructuredMagicTheGatheringService implements AbstractMagicTheGatheringService {
018            private static StructuredMagicTheGatheringService instance;
019            private JsonMagicTheGatheringService jsonInstance;
020            private Gson gson;
021            /**
022             * **For internal use only!** Protected Constructor guards against instantiation.
023            
024             * @return 
025             */
026            protected  StructuredMagicTheGatheringService() {
027                    this.jsonInstance = JsonMagicTheGatheringService.getInstance();
028                    this.gson = new Gson();
029            }
030            
031            /**
032             * Retrieves the singleton instance.
033            
034             * @return StructuredMagicTheGatheringService
035             */
036            public static StructuredMagicTheGatheringService getInstance() {
037                    if (instance == null) {
038                            synchronized (StructuredMagicTheGatheringService.class) {
039                                    if (instance == null) {
040                                            instance = new StructuredMagicTheGatheringService();
041                                    }
042                            }
043                            
044                    }
045                    return instance;
046            }
047            
048            /**
049             * Establishes a connection to the online service. Requires an internet connection.
050            
051             */
052            @Override
053            public void connect() {
054                    jsonInstance.connect();
055            }
056            
057            /**
058             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
059            
060             */
061            @Override
062            public void disconnect() {
063                    jsonInstance.disconnect();
064            }
065            
066            /**
067             * Retrieves a card by looking up its ID.
068             * @param id The unique id number of the card.
069             * @return HashMap<String, Object>
070             */
071            public HashMap<String, Object> getCard(int id) throws Exception {
072                    return gson.fromJson(jsonInstance.getCard(id), LinkedHashMap.class);
073            }
074            
075            /**
076             * Retrieves a card by looking up its ID.
077             * @param id The unique id number of the card.
078             * @param callback The listener that will be given the data (or error)
079             */
080            public void getCard(int id, final StructuredGetCardListener callback) {
081                    
082                    jsonInstance.getCard(id, new JsonGetCardListener() {
083                        @Override
084                        public void getCardFailed(Exception exception) {
085                            callback.getCardFailed(exception);
086                        }
087                        
088                        @Override
089                        public void getCardCompleted(String data) {
090                            callback.getCardCompleted(gson.fromJson(data, LinkedHashMap.class));
091                        }
092                    });
093                    
094            }
095            
096            /**
097             * Searches the database for cards with the keyword in the card's name.
098             * @param keyword The search term to find this card.
099             * @return HashMap<String, Object>
100             */
101            public ArrayList searchCards(String keyword) throws Exception {
102                    return gson.fromJson(jsonInstance.searchCards(keyword), ArrayList.class);
103            }
104            
105            /**
106             * Searches the database for cards with the keyword in the card's name.
107             * @param keyword The search term to find this card.
108             * @param callback The listener that will be given the data (or error)
109             */
110            public void searchCards(String keyword, final StructuredSearchCardsListener callback) {
111                    
112                    jsonInstance.searchCards(keyword, new JsonSearchCardsListener() {
113                        @Override
114                        public void searchCardsFailed(Exception exception) {
115                            callback.searchCardsFailed(exception);
116                        }
117                        
118                        @Override
119                        public void searchCardsCompleted(String data) {
120                            callback.searchCardsCompleted(gson.fromJson(data, ArrayList.class));
121                        }
122                    });
123                    
124            }
125            
126    }
