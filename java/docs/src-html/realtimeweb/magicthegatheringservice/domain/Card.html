341            /**
342             * Internal constructor to create a Card from a Json representation.
343             * @param json The raw json data that will be parsed.
344             * @param gson The Gson parser. See <a href='https://code.google.com/p/google-gson/'>https://code.google.com/p/google-gson/</a> for more information.
345             * @return 
346             */
347            @SuppressWarnings("unchecked")
348            public  Card(JsonObject json, Gson gson) {
349                    this.id = json.get("id").getAsString();
350                    this.name = json.get("name").getAsString();
351                    this.manaCost = gson.fromJson(json.get("mana").getAsJsonArray(), ArrayList.class);
352                    this.convertedManaCost = json.get("cmc").getAsString();
353                    this.types = gson.fromJson(json.get("type").getAsJsonArray(), ArrayList.class);
354                    this.texts = gson.fromJson(json.get("text").getAsJsonArray(), ArrayList.class);
355                    if (json.get("flavor").getAsJsonArray().size() > 0) {
356                            this.flavors = gson.fromJson(json.get("flavor").getAsJsonArray().get(0).getAsJsonArray(), ArrayList.class);
357                    } else {
358                            this.flavors = new ArrayList<String>();
359                    }
360                    this.power = json.get("power").getAsString();
361                    this.toughness = json.get("power").getAsString();
362                    this.watermark = json.get("watermark").getAsString();
363                    this.set = json.get("set").getAsString();
364                    this.rarity = json.get("rarity").getAsString();
365                    JsonArray js = json.get("prints").getAsJsonArray();
366                    this.allSets = new ArrayList<Print>(); 
367                    for (int i = 0; i < js.size(); i += 1) {
368                            this.allSets.add(new Print(js.get(i).getAsJsonObject(), gson));
369                    }
370                    this.number = json.get("number").getAsString();
371                    this.artist = json.get("artist").getAsString();
372                    this.rating = json.get("rating").getAsString();
373                    this.votes = json.get("votes").getAsString();
374            }
