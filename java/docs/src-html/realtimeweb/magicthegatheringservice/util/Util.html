037            /**
038             * Concatenate a list of objects into a single string (using their
039             * "toString" method), joining them with a ",".
040             * 
041             * @param objects
042             *            A list of objects to concatenate.
043             * @return
044             */
045            @SuppressWarnings("rawtypes")
046            public static String join(List objects) {
047                    return join(",", objects);
048            }
049    
050            /**
051             * Concatenate a list of objects into a single string (using their
052             * "toString" method), joining them with a specified glue string.
053             * 
054             * @param glue
055             *            A string to use to join the objects.
056             * @param objects
057             *            A list of objects to concatenate.
058             * @return
059             */
060            @SuppressWarnings("rawtypes")
061            public static String join(String glue, List objects) {
062                    StringBuilder sb = new StringBuilder();
063                    for (Object o : objects) {
064                            if (sb.length() > 0)
065                                    sb.append(glue);
066                            sb.append(o.toString());
067                    }
068                    return sb.toString();
069            }

071            /**
072             * Change the precision of a double value, according to the new mantissa.
073             * 
074             * @param d
075             *            The double value you want re-precisioned.
076             * @param mantissa
077             *            The number of digits after the decimal place.
078             * @return
079             */
080            public static double formatDecimals(double d, int mantissa) {
081                    String m = "";
082                    for (int i = 0; i < mantissa; i++) {
083                            m += "#";
084                    }
085                    DecimalFormat decimalForm = new DecimalFormat("#." + m);
086                    return Double.valueOf(decimalForm.format(d));
087            }

089            /**
090             * Convert an InputStream to a string.
091             * http://stackoverflow.com/a/309718
092             * @param is
093             * @param bufferSize
094             * @return
095             * @throws IOException 
096             */
097            public static String slurp(final InputStream is, final int bufferSize) throws IOException
098            {
099              final char[] buffer = new char[bufferSize];
100              final StringBuilder out = new StringBuilder();
101                final Reader in = new InputStreamReader(is, "UTF-8");
102                try {
103                  for (;;) {
104                    int rsz = in.read(buffer, 0, buffer.length);
105                    if (rsz < 0)
106                      break;
107                    out.append(buffer, 0, rsz);
108                  }
109                }
110                finally {
111                  in.close();
112                }
113              return out.toString();
114            }

117            public static String get(String url, HashMap<String, String> query_args) throws IllegalStateException, IOException, URISyntaxException {
118                    DefaultHttpClient httpclient = new DefaultHttpClient();
119                    HttpGet service= new HttpGet(url);
120                    
121                    // Build query string
122                    URIBuilder ub = new URIBuilder(service.getURI());
123                    for (Entry<String, String> qa : query_args.entrySet()) {
124                            ub.addParameter(qa.getKey(), qa.getValue());
125                    }
126                    URI uri = ub.build();
127                    ((HttpRequestBase) service).setURI(uri);
128                    
129                    // Execute get request
130                    HttpResponse response = httpclient.execute(service);
131                    return EntityUtils.toString(response.getEntity());
132            }
133            
134            public static String post(String url, Map<String, String> body_args) throws IllegalStateException, IOException, URISyntaxException {
135                    DefaultHttpClient httpclient = new DefaultHttpClient();
136                    HttpPost service= new HttpPost(url);
137                    
138                    // Build body
139                    List<NameValuePair> nameValuePairs = new ArrayList<NameValuePair>();
140                    for (Entry<String, String> ba : body_args.entrySet()) {
141                            nameValuePairs.add(new BasicNameValuePair(ba.getKey(), ba.getValue()));
142                    }
143                    service.setEntity(new UrlEncodedFormEntity(nameValuePairs));
144                    
145                    // Execute post request
146                    HttpResponse response = httpclient.execute(service);
147                    return EntityUtils.toString(response.getEntity());
148            }

150            @SuppressWarnings("rawtypes")
151            public static String hashRequest(String url, Map<String, String> query_arguments) {
152                    String queryStrings = "";
153                    if (query_arguments != null) {
154                            queryStrings = (new TreeMap(query_arguments)).toString();
155                    }
156                    return url + "%" + queryStrings;
157            }
