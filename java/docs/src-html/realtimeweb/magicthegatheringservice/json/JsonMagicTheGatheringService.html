001    package realtimeweb.magicthegatheringservice.json;
002    
003    import realtimeweb.magicthegatheringservice.main.AbstractMagicTheGatheringService;
004    import java.util.HashMap;
005    import realtimeweb.magicthegatheringservice.util.Util;
006    
007    /**
008     * Used to get data as a raw string.
009     */
010    public class JsonMagicTheGatheringService implements AbstractMagicTheGatheringService {
011            private static JsonMagicTheGatheringService instance;
012            protected boolean local;
013            private ClientStore clientStore;
014            /**
015             * **For internal use only!** Protected Constructor guards against instantiation.
016            
017             * @return 
018             */
019            protected  JsonMagicTheGatheringService() {
020                    disconnect();
021                    this.clientStore = new ClientStore();
022            }
023            
024            /**
025             * Retrieves the singleton instance.
026            
027             * @return JsonMagicTheGatheringService
028             */
029            public static JsonMagicTheGatheringService getInstance() {
030                    if (instance == null) {
031                            synchronized (JsonMagicTheGatheringService.class) {
032                                    if (instance == null) {
033                                            instance = new JsonMagicTheGatheringService();
034                                    }
035                            }
036                            
037                    }
038                    return instance;
039            }
040            
041            /**
042             * Establishes a connection to the online service. Requires an internet connection.
043            
044             */
045            @Override
046            public void connect() {
047                    this.local = false;
048            }
049            
050            /**
051             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
052            
053             */
054            @Override
055            public void disconnect() {
056                    this.local = true;
057            }
058            
059            /**
060             * **For internal use only!** The ClientStore is the internal cache where offline data is stored.
061            
062             * @return ClientStore
063             */
064            public ClientStore getClientStore() {
065                    return this.clientStore;
066            }
067            
068            /**
069             * Retrieves a card by looking up its ID.
070             * @param id The unique id number of the card.
071             * @return String
072             */
073            public String getCard(int id) throws Exception {
074                    String url = String.format("http://mtgapi.com/api/v1/fetch/id/%s", String.valueOf(id));
075                    HashMap<String, String> parameters = new HashMap<String, String>();
076                    if (this.local) {
077                            return clientStore.getData(Util.hashRequest(url, parameters));
078                    }
079                    String jsonResponse = "";
080                    try {
081                        jsonResponse = Util.get(url, parameters);
082                        if (jsonResponse.startsWith("<")) {
083                            throw new Exception(jsonResponse);
084                        }
085                        return jsonResponse;
086                    } catch (Exception e) {
087                        throw new Exception(e.toString());
088                    }
089            }
090            
091            /**
092             * Retrieves a card by looking up its ID.
093             * @param id The unique id number of the card.
094             * @param callback The listener that will be given the data (or error).
095             */
096            public void getCard(final int id, final JsonGetCardListener callback) {
097                    
098                    Thread thread = new Thread() {
099                        @Override
100                        public void run() {
101                            try {
102                                callback.getCardCompleted(JsonMagicTheGatheringService.getInstance().getCard(id));
103                            } catch (Exception e) {
104                                callback.getCardFailed(e);
105                            }
106                        }
107                    };
108                    thread.start();
109                    
110            }
111            
112            /**
113             * Searches the database for cards with the keyword in the card's name.
114             * @param keyword The search term to find this card.
115             * @return String
116             */
117            public String searchCards(String keyword) throws Exception {
118                    String url = String.format("http://mtgapi.com/api/v1/fetch/search/%s", String.valueOf(keyword));
119                    HashMap<String, String> parameters = new HashMap<String, String>();
120                    if (this.local) {
121                            return clientStore.getData(Util.hashRequest(url, parameters));
122                    }
123                    String jsonResponse = "";
124                    try {
125                        jsonResponse = Util.get(url, parameters);
126                        if (jsonResponse.startsWith("<")) {
127                            throw new Exception(jsonResponse);
128                        }
129                        return jsonResponse;
130                    } catch (Exception e) {
131                        throw new Exception(e.toString());
132                    }
133            }
134            
135            /**
136             * Searches the database for cards with the keyword in the card's name.
137             * @param keyword The search term to find this card.
138             * @param callback The listener that will be given the data (or error).
139             */
140            public void searchCards(final String keyword, final JsonSearchCardsListener callback) {
141                    
142                    Thread thread = new Thread() {
143                        @Override
144                        public void run() {
145                            try {
146                                callback.searchCardsCompleted(JsonMagicTheGatheringService.getInstance().searchCards(keyword));
147                            } catch (Exception e) {
148                                callback.searchCardsFailed(e);
149                            }
150                        }
151                    };
152                    thread.start();
153                    
154            }
155            
156    }
